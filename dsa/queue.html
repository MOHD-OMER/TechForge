Python — Queue & Deque 
from collections import deque

q = deque()
q.append(10)       # enqueue rear  — O(1)
q.append(20)
q.append(30)
front = q[0]       # peek front    — O(1)
val = q.popleft()  # dequeue front — O(1)

# Deque (double-ended)
dq = deque()
dq.appendleft(5)   # insert at front
dq.append(10)      # insert at rear
dq.popleft()       # remove front
dq.pop()           # remove rear
JavaScript — Queue (O(1) head pointer) 
// Naive: array (shift is O(n) — fine for small n)
const q = [];
q.push(10);        // enqueue rear  — O(1)
q.push(20);
q.push(30);
const front = q[0]; // peek front   — O(1)
const val = q.shift(); // dequeue   — O(n)!

// Proper O(1) queue — head pointer, no shifting
class Queue {
  #items = []; #head = 0;
  enqueue(v) { this.#items.push(v); }        // O(1)
  dequeue() { return this.#items[this.#head++]; } // O(1)
  get front() { return this.#items[this.#head]; }
}
Java — Queue & Deque (ArrayDeque) 
// ArrayDeque as queue
Deque<Integer> q = new ArrayDeque<>();
q.offer(10);       // enqueue rear  — O(1)
q.offer(20);
q.offer(30);
int front = q.peek();  // peek front    — O(1)
int val = q.poll();    // dequeue front — O(1)

// Deque (double-ended)
Deque<Integer> dq = new ArrayDeque<>();
dq.offerFirst(5);   // insert at front
dq.offerLast(10);   // insert at rear
dq.pollFirst();     // remove front
dq.pollLast();      // remove rear
