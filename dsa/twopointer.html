def two_sum_sorted(arr, target):
    lo, hi = 0, len(arr) - 1
    while lo < hi:
        total = arr[lo] + arr[hi]
        if   total == target: return [lo, hi]
        elif total < target:  lo += 1
        else:                hi -= 1
    return []
