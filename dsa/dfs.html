def dfs(graph, start):
    visited = set()
    stack = [start]

    while stack:
        node = stack.pop()
        if node in visited:
            continue
        visited.add(node)
        print(node, end=" ")

        for neighbor in reversed(graph[node]):
            if neighbor not in visited:
                stack.append(neighbor)
