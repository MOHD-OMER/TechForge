def get_height(node):
    if not node: return 0
    return 1 + max(get_height(node.left), get_height(node.right))

def balance_factor(node):
    return get_height(node.left) - get_height(node.right)

def rotate_right(y):
    x = y.left; T2 = x.right
    x.right = y; y.left = T2
    return x
