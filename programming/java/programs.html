#1Check Prime NumberBasic

static boolean isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= Math.sqrt(n); i++)
        if (n % i == 0) return false;
    return true;
}
isPrime(17);   // true

#21Binary SearchIntermediate
Java
static int binarySearch(int[] a, int target) {
    int lo = 0, hi = a.length - 1;
    while (lo <= hi) {
        int mid = (lo + hi) >>> 1;
        if (a[mid] == target) return mid;
        if (a[mid] < target) lo = mid + 1; else hi = mid - 1;
    }
    return -1;
}
#22Bubble SortIntermediate
Java
static void bubbleSort(int[] a) {
    for (int i = 0; i < a.length; i++)
        for (int j = 0; j < a.length - i - 1; j++)
            if (a[j] > a[j + 1]) {
                int t = a[j]; a[j] = a[j + 1]; a[j + 1] = t;
            }
}

static int[] rotate(int[] a, int k) {
    int n = a.length; k %= n;
    int[] r = new int[n];
    for (int i = 0; i < n; i++) r[(i + k) % n] = a[i];
    return r;
}

#32Merge Two Sorted ArraysIntermediate
Java
static int[] merge(int[] a, int[] b) {
    int[] r = new int[a.length + b.length];
    int i = 0, j = 0, k = 0;
    while (i < a.length && j < b.length)
        r[k++] = a[i] < b[j] ? a[i++] : b[j++];
    while (i < a.length) r[k++] = a[i++];
    while (j < b.length) r[k++] = b[j++];
    return r;
}

#34Two Sum (HashMap)Intermediate
Java
static int[] twoSum(int[] a, int target) {
    Map<Integer, Integer> seen = new HashMap<>();
    for (int i = 0; i < a.length; i++) {
        int need = target - a[i];
        if (seen.containsKey(need)) return new int[]{seen.get(need), i};
        seen.put(a[i], i);
    }
    return new int[0];
}

#36Merge SortAdvanced
Java
static void mergeSort(int[] a, int lo, int hi) {
    if (lo >= hi) return;
    int mid = (lo + hi) >>> 1;
    mergeSort(a, lo, mid);
    mergeSort(a, mid + 1, hi);
    int[] tmp = new int[hi - lo + 1];
    int i = lo, j = mid + 1, k = 0;
    while (i <= mid && j <= hi) tmp[k++] = a[i] <= a[j] ? a[i++] : a[j++];
    while (i <= mid) tmp[k++] = a[i++];
    while (j <= hi) tmp[k++] = a[j++];
    System.arraycopy(tmp, 0, a, lo, tmp.length);
}
#37Quick SortAdvanced
Java
static void quickSort(int[] a, int lo, int hi) {
    if (lo >= hi) return;
    int pivot = a[hi], i = lo - 1;
    for (int j = lo; j < hi; j++)
        if (a[j] < pivot) { i++; int t = a[i]; a[i] = a[j]; a[j] = t; }
    int t = a[i + 1]; a[i + 1] = a[hi]; a[hi] = t;
    quickSort(a, lo, i);
    quickSort(a, i + 2, hi);
}
#38Generic Stack (OOP + Generics)Advanced
Java
class Stack<T> {
    private List<T> items = new ArrayList<>();
    void push(T x) { items.add(x); }
    T pop() {
        if (items.isEmpty()) throw new RuntimeException("empty");
        return items.remove(items.size() - 1);
    }
    boolean isEmpty() { return items.isEmpty(); }
}
#39Bank Account System (OOP)Advanced
Java
class Account {
    private double balance;
    void deposit(double amt) {
        if (amt <= 0) throw new IllegalArgumentException("bad amount");
        balance += amt;
    }
    void withdraw(double amt) {
        if (amt > balance) throw new IllegalStateException("insufficient funds");
        balance -= amt;
    }
    double getBalance() { return balance; }
}
#40Concurrent Counter (Threads)Advanced
Java
import java.util.concurrent.atomic.*;

AtomicInteger counter = new AtomicInteger(0);
List<Thread> threads = new ArrayList<>();
for (int i = 0; i < 4; i++) {
    Thread t = new Thread(() -> {
        for (int j = 0; j < 1000; j++) counter.incrementAndGet();
    });
    threads.add(t); t.start();
}
for (Thread t : threads) t.join();
counter.get();   // 4000 — always correct, no lost updates
